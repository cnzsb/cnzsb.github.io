---

```bash
# --stream
lerna create <package-name>
lerna bootstrap --stream
lerna add <package> [-D] --scope=@<package-name>
lerna run <package-scripts> --scrope=@<package-name> --stream
lerna version
lerna changed
lerna publish -m "chore(request): publish" --no-push
```
